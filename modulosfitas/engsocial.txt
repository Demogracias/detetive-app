        // Renderização da Navegação
        function renderNav() {
            navList.innerHTML = dossiers.map(d => `

        // Efeito de Máquina de Escrever para a Intro
        let typeWriterTimeout;
        function runTypewriter() {
            const target = document.getElementById('typewriter-target');
            const hiddenContent = document.getElementById('hidden-intro-content');
            if(!target) return;
            
            const text = "Acesso autorizado. Descriptografando arquivos operacionais...";
            target.innerHTML = '';
            let i = 0;
            
            clearTimeout(typeWriterTimeout);
            
            function type() {
                if (i < text.length) {
                    target.innerHTML += text.charAt(i);
                    i++;
                    typeWriterTimeout = setTimeout(type, 30); // velocidade digitação
                } else {
                    target.classList.remove('typing-cursor');
                    setTimeout(() => {
                        hiddenContent.classList.remove('hidden');
                        setTimeout(() => {
                            hiddenContent.classList.remove('opacity-0');
                        }, 50);
                    }, 400);
                }
            }
            type();
        }

        // Carregamento de Conteúdo
        function loadDossier(id) {
            currentDossier = id;
            renderNav();
            
            const dossier = dossiers.find(d => d.id === id);
            
            // Efeito de glitch/piscar ao trocar
            mainContent.style.opacity = 0;
            
            setTimeout(() => {
                mainContent.innerHTML = dossier.content;
                mainContent.style.opacity = 1;
                
                if(id === 'intro') {
                    runTypewriter();
                }
            }, 150);
        }
